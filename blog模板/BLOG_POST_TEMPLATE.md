# ========== 必填字段 ==========

# 文章标题（必填）
# - 支持中文和 Emoji
# - 会显示在文章页面和文章卡片上
title: "在这里填写你的文章标题"

# 发布日期（必填）
# - 格式：YYYY-MM-DD（年-月-日）
# - 示例：2025-01-15
published: 2025-01-15


# ========== 推荐字段 ==========

# 文章描述（推荐填写）
# - 会显示在文章卡片上和搜索结果中
# - 建议 50-160 个字符
# - 简要概括文章内容
description: "这里填写文章的简短描述，会显示在文章卡片上"

# 文章标签（推荐填写）
# - 用于文章分类和搜索
# - 使用数组格式：[标签1, 标签2, 标签3]
# - 建议 2-5 个标签
# - 示例：[JavaScript, 前端开发, 教程]
tags: [标签1, 标签2, 标签3]

# 文章分类（推荐填写）
# - 用于组织文章
# - 只能有一个分类
# - 示例："技术"、"生活"、"教程"等
category: "分类名称"


# ========== 可选字段 ==========

# 草稿状态（可选）
# - true: 草稿状态，不会在博客中显示
# - false 或不填: 正式发布
# - 默认：false（发布状态）
draft: false

# 文章封面图（可选）
# - 封面图片的 URL
# - 建议尺寸：1200x630px 或 16:9 比例
# - 支持 JPG、PNG、WebP 格式
# - 如果不填写，系统会自动生成封面
# cover: "https://example.com/cover-image.jpg"

# 作者名称（可选）
# - 如果是转载文章，填写原作者名称
# - 如果是原创文章，可以不填（默认使用配置中的用户名）
# author: "作者名称"


# ========== 转载文章专用字段 ==========
# 如果这是转载文章，请填写以下字段
# 如果是原创文章，可以删除这部分

# 版权许可证名称（转载时填写）
# - 常见许可证：
#   - "CC BY-NC-SA 4.0" (署名-非商业性使用-相同方式共享)
#   - "CC BY 4.0" (署名)
#   - "MIT License"
#   - "Unlicensed" (未指定许可证)
# licenseName: "CC BY-NC-SA 4.0"

# 版权许可证链接（转载时填写）
# - 许可证的完整 URL
# - 与 licenseName 配合使用
# licenseUrl: "https://creativecommons.org/licenses/by-nc-sa/4.0/"

# 原文链接（转载时填写）
# - 原文章的 URL
# - 让读者可以访问原文
# sourceLink: "https://example.com/original-article"
